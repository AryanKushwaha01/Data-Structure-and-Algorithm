class Solution {
public:
    void Solve(Node* root, vector<int>& ans){
        if(root == NULL) return;

        ans.push_back(root -> val);
        
        for(Node* child : root -> children){
            Solve(child, ans);
        }
    }
    vector<int> preorder(Node* root) {
        vector<int> ans;
        Solve(root, ans);

        return ans;
    }
};
