void sortColors(vector<int>& nums) {

        //First Using Direct Way
        // sort(nums.begin(), nums.end());

        //Second Way
        
        for(int i = 0; i < nums.size(); i++){
            for(int j = 0; j < nums.size()-i-1; j++){
                if(nums[j] > nums[j+1]){
                    swap(nums[j], nums[j+1]);
                }
            }
        }
    }
