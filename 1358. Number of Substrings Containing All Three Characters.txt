int numberOfSubstrings(string s) {
        int n = s.size();
        int cnt = 0;
        unordered_map<char, int> m;
        int left = 0;
        for(int right = 0; right < n; right++){
            m[s[right]]++;
            while(m.size() == 3){
		cnt = cnt + (n-right);
                m[s[left]]--;
                if(m[s[left]] == 0){
                    m.erase(s[left]);
                }
                left++;
            }
        }
        return cnt;
    }
