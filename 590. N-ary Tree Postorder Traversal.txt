class Solution {
public:
    void Solve(Node* root, vector<int> &ans){
        if(root == NULL) return;
        for(Node* child : root -> children){
            Solve(child, ans);
        }
        ans.push_back(root->val);
    }
    vector<int> postorder(Node* root) {
        vector<int> ans;
        Solve(root, ans);

        return ans;
    }
};
