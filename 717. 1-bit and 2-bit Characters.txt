 bool isOneBitCharacter(vector<int>& bits) {
        int n = bits.size();
        if(n == 0) return 1;
        int i = 0;
        
        while(i < n-1){
            if(bits[i] == 1){
                i = i+2;
            }else{
                i = i + 1;
            }

        }
        return i == n - 1;
    }
