void Solve(int i, vector<int>& candidates, int target, vector<int>& c, vector<vector<int>>& ans){
        int n = candidates.size();
        if(target == 0){
            ans.push_back(c);
            return;
        }
        if(i >= n){
            return;
        }
        if(target < 0) return;
	 
	if(candidates[i] <= target){
            c.push_back(candidates[i]);
            Solve(i, candidates, target-candidates[i], c, ans);
            c.pop_back();
        }
        Solve(i+1, candidates, target, c, ans);
    }

 vector<vector<int>> combinationSum(vector<int>& candidates, int target) {
        
        vector<int> c;
        vector<vector<int>> ans;

        Solve(0,candidates, target, c, ans);

        return ans;
    }
