int numSub(string s) {
        int n = s.size();
        long long ans = 0;
        long long final = 0;
        
        for(int i = 0; i < n; i++){
            if(s[i] == '1'){
                ans++;
            }
	      else{
                final = (final + ans*(ans+1)/2)% 1000000007;
                ans = 0;
            }
        }
        final = (final + ans*(ans+1)/2)% 1000000007;
        return final% 1000000007;
    }
