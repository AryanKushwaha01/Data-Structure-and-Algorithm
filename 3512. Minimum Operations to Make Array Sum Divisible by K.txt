int minOperations(vector<int>& nums, int k) {
        int n = nums.size();
        int sum = 0;
        int cnt = 0;
        for(int i = 0; i < n; i++){
            sum = sum + nums[i];
        }
        if(sum%k == 0){
            return 0;
        }
        if(sum%k != 0){
            cnt = sum%k;
        }
        return cnt;
        
    }
