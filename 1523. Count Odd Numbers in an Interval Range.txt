int countOdds(int low, int high) {
        // int cnt = 0;
        // for(int i = low; i <= high; i++){
        //     if(i%2 != 0) cnt++;
        // }
        // return cnt;
        int l = 0;
        int h = 0;
        if(low%2 == 0){
            l = (low/2);
        }
	 else{
            l = (low+1)/2-1;
        }
        if(high%2 == 0){
            h = (high/2);
        }
        else{
            h = (high+1)/2;
        }
        return h-l;
    }
