 bool canPlaceFlowers(vector<int>& nums, int n) {
        int n1 = nums.size();
        if(n == 0) return 1;

        for(int i = 0; i < n1; i++){
            if(nums[i] == 0){
                if((i == 0 || nums[i-1] == 0) && (i == n1-1 || nums[i+1] == 0) && (n > 0)){
                    nums[i] = 1;
                    n--;
                }
                if(n == 0){
                    return 1;
                }
            }
        }
        return 0;
    }
