int countHomogenous(string s) {
        int n = s.size();
        long long ans = 0;
        long long final = 0;

        for(int i = 0; i < n; i++){
            if(s[i] != s[i+1] && (i >= 0 && i < n)){
                ans++;
                final = (final + ans*(ans+1)/2)%1000000007;
                ans = 0;
            }
	   else{
                ans++;
            }
        }
        final = (final + ans*(ans+1)/2)% 1000000007;
        return final% 1000000007;
    }
};
