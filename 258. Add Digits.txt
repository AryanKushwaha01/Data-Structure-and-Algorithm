int addDigits(int n) {
        int sum = 0;
        if(n < 10) return n;
        while(n > 9){
            sum = 0;
            while(n > 0){
                int digit = n%10;
                sum = sum + digit;
                n = n/10;
            }
            n = sum;
        }
        return sum;
    }
