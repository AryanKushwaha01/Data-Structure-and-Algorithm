int minDepth(TreeNode* root) {
        if(root == nullptr) return 0;
        int left = 0;
        int right = 0;
        left = minDepth(root -> left);
        right = minDepth(root -> right);
        if(root -> left == NULL) return 1+ right;
        if(root -> right == NULL) return 1 + left;
        return 1 + min(left, right);
    }
