int maxPower(string s) {
        int n = s.size();
        int ans = 0;
        int final = 0;

        for(int i = 0; i < n; i++){
            if(s[i] != s[i+1] && (i >=0 && i < n)){
                ans++;
                final = max(final, ans);
                ans = 0;
            }
	       else{
                ans++;
            }
        }
        return final;
    }
