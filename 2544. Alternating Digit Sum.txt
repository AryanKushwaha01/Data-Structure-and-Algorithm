int alternateDigitSum(int n1) {
        if(n1 < 10) return n1;
        string s = to_string(n1);
        reverse(s.begin(), s.end());
        int sum = 0;
        int i = 0;
        int n = stoi(s);
        while(n > 0){
            i++;
            int digit = n%10;
            if(i%2 != 0){
                sum = sum + digit;
            }
	    else{
                sum = sum - digit;
            }
            n = n/10;
        }
    }
