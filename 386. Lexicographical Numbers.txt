void rec(int prefix, vector<int>& ans, int n){
        if(prefix > n) return;
        if(prefix != 0) ans.push_back(prefix);

        for(int d = (prefix == 0?1:0); d<=9; d++){
            if(prefix<=n){
                rec(prefix*10+d, ans, n);
            }
        }
    }
    vector<int> lexicalOrder(int n) {
        vector<int> ans;
        rec(0,ans, n);
        return ans;
    }
