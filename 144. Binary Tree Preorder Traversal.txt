 vector<int> Solve(TreeNode* root, vector<int>& ans){
        if(root == NULL) return {};

        ans.push_back(root -> val);
        Solve(root -> left, ans);
        Solve(root -> right, ans);

        return ans;
    }
vector<int> preorderTraversal(TreeNode* root) {
        vector<int> ans;
        Solve(root, ans);

        return ans;
    }
