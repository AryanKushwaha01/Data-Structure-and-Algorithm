int longestSubarray(vector<int>& nums) {
        int n = nums.size();
        unordered_map<int, int> m;
        int maxLen = 0;
        int left = 0;

        for(int right = 0; right < n; right++){
            m[nums[right]]++;
            if(m[0] > 1){
                m[nums[left]]--;
                left++;
            }
	     maxLen = max(maxLen, right - left);
        }

        return maxLen;
    }
