int sumOfUnique(vector<int>& nums) {
        int n = nums.size();

        unordered_map<int, int> m;

        for(int i : nums){
            m[i]++;
        }
        int sum = 0;

        for(int i = 0; i < n; i++){
            if(m[nums[i]] == 1){
                sum = sum + nums[i];
            }
        }
        return sum;
    }
